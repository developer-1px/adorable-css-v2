Use in Your Project

// adorable.config.ts
import { generateTokenScales } from '@adorable-css/design-system/tokens';

export const tokens = generateTokenScales({
  typography: {
    base: 16,
    ratio: 1.2,
    unit: 'rem',
    clamp: { min: 10, max: 128 }
  },
  spacing: {
    baseUnit: 4,
    scale: 'progressive',
    unit: 'rem'
  }
});
